#### -- Get Classes / Levels of a categorical Variable -- ####
def getclasses(df):
    i = 0
    F = pd.DataFrame()
    for c in df.select_dtypes('object').columns:
            t_i = pd.DataFrame(df[c].unique()).rename(columns={0: c}).sort_values(c).reset_index()
            F = pd.concat([F, t_i.iloc[:,1]], axis=1)
            i += 1
            
    F = F.replace(np.nan, '', regex=True).T

    NClass = []
    for r in range(0,len(F)):
        i = 0
        robs = F.iloc[r,:]
        for c in robs:
            if (c == ""):
                continue
            else:
                i +=1
        NClass.append(i)

    df2 = F
    df2['NClass'] = NClass

    cols = df2.columns.tolist()
    cols = cols[-1:] + cols[:-1]
    df2 = df2[cols]
    return df2.sort_values('NClass', ascending = False)

#### -- Fields with Missing Values -- ####
def missing_values_table(df):
        mis_val = df.isnull().sum()
        mis_val_percent = 100 * df.isnull().sum() / len(df)
        mis_val_table = pd.concat([mis_val, mis_val_percent], axis=1)
        mis_val_table_ren_columns = mis_val_table.rename(
        columns = {0 : 'Missing Values', 1 : '% of Total Values'})
        
        mis_val_table_ren_columns = mis_val_table_ren_columns[
            mis_val_table_ren_columns.iloc[:,0] != 0].sort_values('% of Total Values', ascending=False).round(1)
        
        print ("Your selected dataframe has " + str(df.shape[1]) + " columns.\n"      
            "There are " + str(mis_val_table_ren_columns.shape[0]) +
              " columns that have missing values.")
        
        # Return the dataframe with missing information
        return mis_val_table_ren_columns
